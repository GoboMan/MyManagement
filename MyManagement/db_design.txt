//-----------------------------------------------------------------------------
//  ユーザー
//-----------------------------------------------------------------------------
user
{
    user_id,                bigint, ai, pk                  # ユーザー
    username,               varchar, 32,                    # ユーザーネーム
    email,                  email,                          # メールアドレス
    password,               varchar, 64,                    # パスワード
    profile_pic,            url,                            # プロフィール写真
    current_streak,         int,                            # 現在連続日数
    max_streak,             int,                            # 最大連続日数
    deleted,                del,                            # 削除フラグ
    created_at,             datetime,                       # 作成日
    updated_at,             datetime,                       # 更新日
}

    language_id,            int,                            # 言語
    theme_id,               int,                            # テーマ
    notification_enabled,   boolean,                        # 通知
    private_mode,           boolean,                        # 非公開モード, 要承認
}
